virtual void HandleStreamFailed ( const StreamIdentifier &  stream)
pure virtual

Stream close event.

Called when the other side forcefully closes the stream. No updates will be delivered for the stream provided as a parameter after this event.

void MyStrategy::HandleStreamFailed(const Stream & stream)
{
std::vector<tbricks::BestPrice>::iterator i;
for (i = m_best_price_streams.begin();
i != m_best_price_streams.end(); ++i) {
if (*i == stream) {
// this stream is closed by the other side, either
// erase it or re-open.
break;
}
}
}
