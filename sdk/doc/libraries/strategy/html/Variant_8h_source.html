474  template <typename T, typename... Args, typename = std::enable_if_t<is_supported_type<T>::value>>
475  T & Emplace(Args &&... args)
476  {
477  if (m_type != TypeInfo::SFT_NONE) {
478  Clear();
479  }
480  m_type = T::GetTypeInfo().GetType();
481  if constexpr (!is_requires_allocation<T>())
482  return *(new (&m_stack) T(std::forward<Args>(args)...));
483  else {
484  return *static_cast<T *>(m_dynamic = new T(std::forward<Args>(args)...));
